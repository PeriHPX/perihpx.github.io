188 Point3 project(const Point3 &b, bool is_unit = false) const {
189 auto l_sq = (is_unit ? 1. : this->length() * this->length());
190 auto dot = this->dot(b);
191 return {dot * d_x / l_sq, dot * d_y / l_sq, dot * d_z / l_sq};
